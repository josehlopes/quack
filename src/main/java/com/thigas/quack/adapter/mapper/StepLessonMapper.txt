package com.thigas.quack.adapter.mapper;

import com.thigas.quack.adapter.dto.StepLessonDTO;
import com.thigas.quack.domain.entity.StepLessonEntity;
import com.thigas.quack.infrastructure.persistence.entity.StepLessonModel;
import org.mapstruct.Mapping;
import org.mapstruct.factory.Mappers;

public interface StepLessonMapper {

    StepLessonMapper INSTANCE = Mappers.getMapper(StepLessonMapper.class);

    // Mapeia StepLessonEntity para StepLessonModel
    @Mapping(source = "lessons", target = "lessons", qualifiedByName = "mapLessonEntitiesToModels")
    StepLessonModel toStepLessonModel(StepLessonEntity stepLessonEntity);

    // Mapeia StepLessonModel para StepLessonEntity
    @Mapping(source = "lessons", target = "lessons", qualifiedByName = "mapLessonModelsToEntities")
    StepLessonEntity toStepLessonEntity(StepLessonModel stepLessonModel);

    // Mapeia StepLessonEntity para StepLessonDTO
    @Mapping(source = "lessons", target = "lessonIds", qualifiedByName = "mapLessonEntitiesToIds")
    StepLessonDTO toStepLessonDTO(StepLessonEntity stepLessonEntity);

    // Mapeia StepLessonDTO para StepLessonEntity
    @Mapping(source = "lessonIds", target = "lessons", qualifiedByName = "mapIdsToLessonEntities")
    StepLessonEntity toStepLessonEntity(StepLessonDTO stepLessonDTO);

}
